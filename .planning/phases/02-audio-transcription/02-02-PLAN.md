```swift
import Foundation

enum STTError: LocalizedError {
    case notLoaded
    case transcriptionFailed(underlying: Error)
    case emptyResult

    var errorDescription: String? {
        switch self {
        case .notLoaded:
            return "El modelo de transcripcion no esta cargado"
        case .transcriptionFailed(let error):
            return "Error de transcripcion: \(error.localizedDescription)"
        case .emptyResult:
            return "La transcripcion no produjo texto"
        }
    }
}
```

**Step 1: Create MyWhisper/STT/STTEngine.swift**

This is the core STT actor. It must:

1. Import WhisperKit.
2. Be declared as `actor STTEngine: STTEngineProtocol`.
3. Hold private properties: `whisperKit: WhisperKit?`, `isLoading: Bool = false`, `_loadProgress: Double = 0.0`.
4. Implement `prepareModel()`:
   - Guard against re-entry: `guard whisperKit == nil, !isLoading else { return }`.
   - Set `isLoading = true`, defer `isLoading = false`.
   - Download model: call `WhisperKit.download(variant: "openai_whisper-large-v3", from: "argmaxinc/whisperkit-coreml", progressCallback: { progress in ... })`.
   - In the progress callback, update `_loadProgress` with `progress.fractionCompleted * 0.5` (download is 50% of total load progress).
   - Create WhisperKitConfig with: `model: "openai_whisper-large-v3"`, `modelFolder: modelFolder` (from download), `computeOptions: .init(audioEncoderCompute: .cpuAndNeuralEngine)`.
   - Initialize WhisperKit: `whisperKit = try await WhisperKit(config)`.
   - Set `_loadProgress = 0.75`.
   - Call `try await whisperKit?.prewarmModels()` and then `try await whisperKit?.loadModels()`.
   - Set `_loadProgress = 1.0`.
5. Implement `transcribe(_ audioArray: [Float]) async throws -> String`:
   - If whisperKit is nil and not loading, call `try await prepareModel()` (handles case where user records before model finishes per CONTEXT.md decision).
   - If whisperKit is still nil after prepare, throw `STTError.notLoaded`.
   - Create DecodingOptions with: `language: "es"` (force Spanish -- no auto-detection per locked decision), `temperature: 0.0`, `temperatureFallbackCount: 3`, `sampleLength: 224`, `noSpeechThreshold: 0.6`.
   - Call `try await whisperKit!.transcribe(audioArray: audioArray, decodeOptions: options)`.
   - Map results: `results.map { $0.text }.joined(separator: " ").trimmingCharacters(in: .whitespaces)`.
   - If result text is empty, throw `STTError.emptyResult`.
   - Return the joined text.
6. Implement computed properties:
   - `var isReady: Bool { whisperKit != nil }`
   - `var loadProgress: Double { _loadProgress }`

**IMPORTANT API notes from RESEARCH.md:**
- The model variant string is `"openai_whisper-large-v3"` and the repo is `"argmaxinc/whisperkit-coreml"`.
- Must call BOTH `prewarmModels()` AND `loadModels()` -- prewarm alone is not sufficient.
- Never call transcribe on MainActor -- actor isolation handles this automatically.
- WhisperKit transcribes fixed 30-second Mel windows internally -- pass the entire buffer, never chunk it.

**Step 2: Create MyWhisperTests/STTEngineTests.swift**

Create unit tests for STT-02 (model readiness and protocol behavior) using a MockSTTEngine. The real STTEngine cannot be unit-tested (requires model download), so we test the protocol contract with a mock to verify that consumers (AppCoordinator) can rely on isReady and transcribe behavior.

```swift
import XCTest
@testable import MyWhisper

/// MockSTTEngine for testing STTEngineProtocol consumers.
/// Verifies that the protocol contract supports model readiness checks,
/// progress reporting, and transcription (STT-02).
final class MockSTTEngine: STTEngineProtocol, @unchecked Sendable {
    var mockTranscription: String = "Hola mundo"
    var shouldThrow = false
    var prepareModelCalled = false
    var transcribeCalled = false
    private var _isReady = false
    private var _loadProgress: Double = 0.0

    var isReady: Bool { _isReady }
    var loadProgress: Double { _loadProgress }

    func prepareModel() async throws {
        prepareModelCalled = true
        _loadProgress = 1.0
        _isReady = true
    }

    func transcribe(_ audioArray: [Float]) async throws -> String {
        transcribeCalled = true
        if shouldThrow {
            throw STTError.transcriptionFailed(underlying: NSError(domain: "test", code: 1))
        }
        if audioArray.isEmpty {
            throw STTError.emptyResult
        }
        return mockTranscription
    }
}

final class STTEngineTests: XCTestCase {

    // MARK: - Model readiness (STT-02)

    func testMockEngineNotReadyBeforePrepare() async {
        let engine = MockSTTEngine()
        let ready = await engine.isReady
        XCTAssertFalse(ready, "Engine should not be ready before prepareModel()")
    }

    func testMockEngineReadyAfterPrepare() async throws {
        let engine = MockSTTEngine()
        try await engine.prepareModel()
        let ready = await engine.isReady
        XCTAssertTrue(ready, "Engine should be ready after prepareModel()")
        XCTAssertTrue(engine.prepareModelCalled)
    }

    func testLoadProgressZeroBeforePrepare() async {
        let engine = MockSTTEngine()
        let progress = await engine.loadProgress
        XCTAssertEqual(progress, 0.0, "Progress should be 0 before model load")
    }

    func testLoadProgressOneAfterPrepare() async throws {
        let engine = MockSTTEngine()
        try await engine.prepareModel()
        let progress = await engine.loadProgress
        XCTAssertEqual(progress, 1.0, "Progress should be 1.0 after model load")
    }

    // MARK: - Transcription (STT-02 protocol contract)

    func testTranscribeReturnsText() async throws {
        let engine = MockSTTEngine()
        try await engine.prepareModel()
        let text = try await engine.transcribe([0.1, 0.2, 0.3])
        XCTAssertEqual(text, "Hola mundo")
        XCTAssertTrue(engine.transcribeCalled)
    }

    func testTranscribeThrowsOnError() async throws {
        let engine = MockSTTEngine()
        engine.shouldThrow = true
        try await engine.prepareModel()
        do {
            _ = try await engine.transcribe([0.1, 0.2])
            XCTFail("Expected transcription error")
        } catch {
            XCTAssertTrue(error is STTError)
        }
    }

    func testTranscribeEmptyBufferThrows() async throws {
        let engine = MockSTTEngine()
        try await engine.prepareModel()
        do {
            _ = try await engine.transcribe([])
            XCTFail("Expected emptyResult error for empty buffer")
        } catch let error as STTError {
            if case .emptyResult = error {
                // expected
            } else {
                XCTFail("Expected STTError.emptyResult, got \(error)")
            }
        }
    }

    // MARK: - STTError tests

    func testSTTErrorDescriptions() {
        let notLoaded = STTError.notLoaded
        XCTAssertNotNil(notLoaded.errorDescription)
        XCTAssertTrue(notLoaded.errorDescription!.contains("modelo"))

        let empty = STTError.emptyResult
        XCTAssertNotNil(empty.errorDescription)
        XCTAssertTrue(empty.errorDescription!.contains("texto"))

        let failed = STTError.transcriptionFailed(underlying: NSError(domain: "test", code: 42))
        XCTAssertNotNil(failed.errorDescription)
        XCTAssertTrue(failed.errorDescription!.contains("transcripcion"))
    }
}
```
