Create `MyWhisper/System/KeychainService.swift` as an enum with static methods (matches NotificationHelper pattern):

```swift
import Security
import Foundation

enum KeychainError: Error {
    case saveFailed(OSStatus)
    case deleteFailed(OSStatus)
}

enum KeychainService {
    private static let service = "com.mywhisper.anthropic-api-key"
    private static let account = "anthropic"

    static func save(_ key: String) throws {
        let data = Data(key.utf8)
        let query: [CFString: Any] = [
            kSecClass: kSecClassGenericPassword,
            kSecAttrService: service,
            kSecAttrAccount: account,
            kSecValueData: data
        ]
        SecItemDelete(query as CFDictionary)  // remove existing first
        let status = SecItemAdd(query as CFDictionary, nil)
        guard status == errSecSuccess else { throw KeychainError.saveFailed(status) }
    }

    static func load() -> String? {
        let query: [CFString: Any] = [
            kSecClass: kSecClassGenericPassword,
            kSecAttrService: service,
            kSecAttrAccount: account,
            kSecReturnData: true,
            kSecMatchLimit: kSecMatchLimitOne
        ]
        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess, let data = result as? Data else { return nil }
        return String(data: data, encoding: .utf8)
    }

    static func delete() throws {
        let query: [CFString: Any] = [
            kSecClass: kSecClassGenericPassword,
            kSecAttrService: service,
            kSecAttrAccount: account
        ]
        let status = SecItemDelete(query as CFDictionary)
        guard status == errSecSuccess || status == errSecItemNotFound else {
            throw KeychainError.deleteFailed(status)
        }
    }
}
```

**File 2: `MyWhisper/Cleanup/HaikuCleanupError.swift`**
```swift
enum HaikuCleanupError: Error, LocalizedError {
    case noAPIKey
    case authFailed          // 401 / 403
    case rateLimited         // 429
    case serverError(Int)    // 500 / 529
    case invalidResponse     // could not parse content[0].text
    case networkError(Error) // URLError.timedOut, no connection, etc.

    var errorDescription: String? {
        switch self {
        case .noAPIKey:       return "No hay clave de API configurada"
        case .authFailed:     return "Clave de API invalida o sin credito"
        case .rateLimited:    return "Limite de solicitudes alcanzado"
        case .serverError(let code): return "Error del servidor Anthropic (\(code))"
        case .invalidResponse: return "Respuesta inesperada de la API"
        case .networkError:   return "Error de red"
        }
    }
}
```

**File 3: `MyWhisper/Cleanup/HaikuCleanupService.swift`**

Actor with:
- `private let session: URLSession` — custom config with `timeoutIntervalForRequest = 5.0` and `timeoutIntervalForResource = 5.0`, `requestCachePolicy = .reloadIgnoringLocalCacheData`
- `private let systemPrompt: String` — the full Spanish cleanup prompt from RESEARCH.md (exact text from the "System Prompt" code example)
- `init(session: URLSession? = nil)` — accepts optional URLSession for test injection
- `func clean(_ rawText: String) async throws -> String`:
  1. Load API key via `KeychainService.load()` — throw `.noAPIKey` if nil
  2. Build URLRequest to `https://api.anthropic.com/v1/messages`
  3. Headers: `x-api-key: {apiKey}`, `anthropic-version: 2023-06-01`, `content-type: application/json`
  4. Body JSON: `model: "claude-haiku-4-5-20251001"`, `max_tokens: estimateMaxTokens(for: rawText)`, `system: systemPrompt`, `messages: [{"role": "user", "content": rawText}]`
  5. `let (data, response) = try await session.data(for: request)`
  6. Switch on HTTP status: 200 → decode AnthropicResponse, extract `content.first(where: { $0.type == "text" })?.text` → return it or throw `.invalidResponse`. 401/403 → throw `.authFailed`. 429 → throw `.rateLimited`. Other → throw `.serverError(code)`
  7. Catch URLError → rethrow as `.networkError(error)`
- `var hasAPIKey: Bool` — returns `KeychainService.load() != nil`
- `func saveAPIKey(_ key: String) async throws` — validate with small test request (max_tokens: 5, messages: [{"role":"user","content":"hola"}]), if 200 then `KeychainService.save(key)`, if not throw appropriate error
- `func removeAPIKey() async throws` — calls `KeychainService.delete()`
- `func estimateMaxTokens(for text: String) -> Int` — `min(max(Int(Double(text.count) / 4.0 * 1.5), 128), 2048)`
- Private struct `AnthropicResponse: Decodable` with `content: [ContentBlock]?` where `ContentBlock` has `type: String` and `text: String?`
