    ```swift
    private func isAnyMediaAppRunning() -> Bool {
        let mediaApps: Set<String> = [
            "com.spotify.client",
            "com.apple.Music",
            "org.videolan.vlc",
            "com.apple.Safari",
            "com.google.Chrome",
            "org.mozilla.firefox"
        ]
        return NSWorkspace.shared.runningApplications
            .compactMap(\.bundleIdentifier)
            .contains { mediaApps.contains($0) }
    }
    ```
