**Paso 2: Reescribir SettingsView.swift**

Reemplazar el contenido completo de `MyWhisper/Settings/SettingsView.swift` con:

```swift
import SwiftUI
import KeyboardShortcuts

struct SettingsView: View {
    @Bindable var viewModel: SettingsViewModel

    var body: some View {
        Form {
            // SECCION 1: GRABACION (per D-01, D-02)
            Section {
                KeyboardShortcuts.Recorder("Atajo de grabación:", name: .toggleRecording)

                Picker("Micrófono:", selection: $viewModel.selectedMicID) {
                    Text("Predeterminado del sistema")
                        .tag(nil as AudioDeviceID?)
                    ForEach(viewModel.availableMics) { device in
                        Text(device.name)
                            .tag(device.id as AudioDeviceID?)
                    }
                }

                Toggle("Pausar reproducción al grabar", isOn: $viewModel.pausePlaybackEnabled)
                Toggle("Maximizar volumen del micrófono", isOn: $viewModel.maximizeMicVolumeEnabled)
            } header: {
                Label("Grabación", systemImage: "mic.fill")
            }

            // SECCION 2: API (per D-01, D-03, D-12)
            Section {
                Button("Configurar clave API...") {
                    viewModel.openAPIKey()
                }
            } header: {
                Label("API", systemImage: "key.fill")
            }

            // SECCION 3: VOCABULARIO (per D-01, D-04, D-07, D-08)
            Section {
                ForEach($viewModel.vocabularyEntries) { $entry in
                    HStack {
                        TextField("Incorrecto", text: $entry.wrong)
                        Image(systemName: "arrow.right")
                            .foregroundColor(.secondary)
                        TextField("Correcto", text: $entry.correct)
                        Button {
                            viewModel.vocabularyEntries.removeAll { $0.id == entry.id }
                        } label: {
                            Image(systemName: "minus.circle.fill")
                                .foregroundColor(.red)
                        }
                        .buttonStyle(.plain)
                    }
                }
                Button {
                    viewModel.vocabularyEntries.append(VocabularyEntry(wrong: "", correct: ""))
                } label: {
                    Label("Agregar corrección", systemImage: "plus")
                }
            } header: {
                Label("Vocabulario", systemImage: "textformat.abc")
            }

            // SECCION 4: SISTEMA (per D-01, D-05)
            Section {
                Toggle("Iniciar al arranque", isOn: $viewModel.launchAtLoginEnabled)
            } header: {
                Label("Sistema", systemImage: "gear")
            }
        }
        .formStyle(.grouped)
        .frame(minWidth: 420)
    }
}
```

Puntos criticos de implementacion:
- El `.frame(minWidth: 420, minHeight: 200)` se cambia a `.frame(minWidth: 420)` sin minHeight — per D-10 y RESEARCH Pitfall 3 para auto-sizing.
- El Picker de microfono usa `.tag(nil as AudioDeviceID?)` y `.tag(device.id as AudioDeviceID?)` — ambos casts a tipo opcional son OBLIGATORIOS per RESEARCH Pitfall 2. Sin estos casts, el Picker no muestra la seleccion activa.
- El boton [-] de vocabulario usa `.buttonStyle(.plain)` con Image `minus.circle.fill` en rojo — per RESEARCH Pitfall 5 y UI-SPEC.
- NO usar `onDelete`, `EditButton`, `@AppStorage`, ni dimensiones fijas de altura.
- Orden dentro de Grabacion: (1) Hotkey recorder, (2) Picker microfono, (3) Toggle pausar, (4) Toggle maximizar — per D-02.
